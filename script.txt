CREATE TABLE cities (
	id INTEGER PRIMARY KEY AUTOINCREMENT,
	name VARCHAR(30),
	code VARCHAR(6),
	region_id INTEGER,
	FOREIGN KEY (region_id) REFERENCES regions(id)
);
